# Script: Route 2 / Early Unova Wild Encounter Farm
# Requisitos:
#   - Posicion inicial: parado encima del Centro Pokemon de Accumula Town
#   - Resolucion: 1920x1080
#   - Un Pokemon con un ataque rapido en posicion 1 del equipo
#   - NO necesita HM Surf
#
# Flujo:
#   setup -> entrar al centro a curar
#   loop  -> bajar a hierba, WALK hasta encuentro,
#            ganar batalla (auto-detectada), volver al centro

setup:
    PRINT iniciando_bot_espera_2_segundos
    SLEEP 2
    HOLD UP
    WAITFOR center
    SLEEP 1
    RELEASE UP

loop:
    # Curar en el Centro Pokemon
    SLEEP 0.5
    PRESS Z
    WAITFOR arrow
    PRESS Z
    WAITFOR yesno
    PRESS Z
    WAITFOR arrow
    PRESS Z
    WAITFOR arrow
    PRESS Z
    WAITFOR arrow
    PRESS Z
    SLEEP 1.5

    # Salir del centro hacia la hierba (Route 2 al sur de Accumula)
    HOLD DOWN
    SLEEP 3.5
    RELEASE DOWN
    SLEEP 0.3

    # Caminar en hierba con patron configurable
    WALK zigzag

    # Si no hubo batalla durante el WALK, esperar una
    WAITFOR lotta
    SPAM Z
    WAITFOR terrain
    STOPSPAM Z
    SLEEP 0.5

    # Volver al Centro Pokemon (subir hasta ver la puerta)
    HOLD UP
    WAITFOR center
    SLEEP 1
    RELEASE UP
