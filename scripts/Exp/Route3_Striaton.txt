# Script: Route 3 - Striaton City (Gym 1 completado)
# Requisitos:
#   - Posicion inicial: encima de la puerta del Centro Pokemon de Striaton City
#   - Resolucion: 1920x1080, juego en ventana
#   - Pokemon nivel ~15-20 con ataque rapido en posicion 1
#   - NO necesita HMs
#
# Flujo:
#   setup -> entrar al centro a curar
#   loop  -> salir, ir a Route 3, WALK en hierba (detecta batalla auto),
#            ganar batalla, volver al centro
#
# Config relevante (config.toml):
#   [battle] MOVE_SLOT = slot del movimiento a usar
#   [walk]   PATTERN = patron de movimiento (zigzag, circle, random...)
#   [walk]   STEP_DURATION = duracion de cada paso
#   [walk]   STEPS = numero de pasos por ciclo

setup:
    PRINT esperando_2_segundos
    SLEEP 2
    HOLD UP
    WAITFOR center
    SLEEP 1
    RELEASE UP

loop:
    # Curar en el Centro Pokemon
    SLEEP 0.5
    PRESS Z
    WAITFOR arrow
    PRESS Z
    WAITFOR yesno
    PRESS Z
    WAITFOR arrow
    PRESS Z
    WAITFOR arrow
    PRESS Z
    WAITFOR arrow
    PRESS Z
    SLEEP 1.5

    # Salir del centro
    HOLD DOWN
    SLEEP 1.0
    RELEASE DOWN
    SLEEP 0.3

    # Ir al este hacia Route 3
    HOLD RIGHT
    SLEEP 4.0
    RELEASE RIGHT
    SLEEP 0.3

    # Caminar en hierba con patron configurable
    # Si encuentra batalla durante el WALK, la maneja automaticamente
    WALK zigzag

    # Si no hubo batalla durante el WALK, esperar una
    WAITFOR lotta
    SPAM Z
    WAITFOR terrain
    STOPSPAM Z
    SLEEP 0.8

    # Volver al Centro Pokemon (oeste largo + subir hasta ver puerta)
    HOLD LEFT
    SLEEP 8
    RELEASE LEFT
    SLEEP 0.3

    HOLD UP
    WAITFOR center
    SLEEP 1
    RELEASE UP
